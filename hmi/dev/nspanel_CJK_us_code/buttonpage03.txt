    Attributes
        ID   : 1
        Scope: local
        Value: 0
    
    Events
        Touch Press Event
            swipestore.en=1 // Start swipestore timer
        
        Touch Release Event
            if(api==0)
            {
                page home_page_id
            }
            swipestore.en=0
            // Touch has ended, x
            if(tch0==0)
            {
                swipec=swipex-tch2
                // From Left to Right
                if(swipec>swipedx)
                {
                    // Navigate back based on active status
                    if(is_buttonpage02==1)
                    {
                        page buttonpage02
                    }else if(is_buttonpage01==1)
                    {
                        page buttonpage01
                    }else
                    {
                        page home_page_id
                    }
                }
                // Right to Left
                swipec2=0-swipedx
                if(swipec<swipec2)
                {
                    // Navigate forward to buttonpage04 or fallback to home
                    if(is_buttonpage04==1)
                    {
                        page buttonpage04
                    }else
                    {
                        page home_page_id
                    }
                }
            }
            // Touch has ended, y
            if(tch1==0)
            {
                swipec=swipey-tch3
                // From Up to Down
                if(swipec>100)
                {
                    if(is_buttonpage02==1&&is_buttonpage01==1)
                    {
                        page buttonpage01
                    }else if(is_buttonpage02==1||is_buttonpage01==1||is_buttonpage04==0)
                    {
                        page home_page_id
                    }else
                    {
                        page buttonpage04
                    }
                }
                // Down to Up
                swipec2=0-swipedy
                if(swipec<-100)
                {
                    // Navigate back up the sequence based on active status
                    if(is_buttonpage04==1)
                    {
                        page home_page_id
                    }else if(is_buttonpage01==1)
                    {
                        page buttonpage01
                    }else if(is_buttonpage02==1)
                    {
                        page buttonpage02
                    }else
                    {
                        page home_page_id
                    }
                }
            }
